import java.util.Scanner;

public class calculateGrade {
    public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
       System.out.print("Enter total number of subjects: ");
        int numSubjects = scanner.nextInt();
         int totalMarks = 0;
    for (int i = 1; i <= numSubjects; i++) {
         System.out.print("Enter marks obtained in " + i + ": ");
            int marks = scanner.nextInt();
                 totalMarks += marks;
        }
         double averagePercent = (double) totalMarks / numSubjects;
            System.out.println("\nTotal Marks obtained: " + totalMarks);
            System.out.println("Average Percentage: " + averagePercent + "%");

        char grade;
            if (averagePercent >= 90) {
             grade = 'A';
        } else if (averagePercent >= 80) {
             grade = 'B';
        } else if (averagePercent >= 70) {
             grade = 'C';
        } else if (averagePercent >= 60) {
            grade = 'D';
        } else {
                grade = 'F';
        }
        System.out.println("You got grade: " + grade);
         scanner.close();
    }
}
